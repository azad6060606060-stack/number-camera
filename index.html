Функция JavaScript для запуска камеры в браузере def take_photo(filename='photo.jpg', quality=0.8): js = Javascript(''' async function takePhoto(quality) { const div = document.createElement('div'); const video = document.createElement('video'); video.style.display = 'block'; const stream = await navigator.mediaDevices.getUserMedia({video: true}); document.body.appendChild(div); div.appendChild(video); video.srcObject = stream; await video.play(); // Ждем клика, чтобы сделать кадр await new Promise((resolve) => { const btn = document.createElement('button'); btn.textContent = 'Распознать цифры'; div.appendChild(btn); btn.onclick = resolve; }); const canvas = document.createElement('canvas'); canvas.width = video.videoWidth; canvas.height = video.videoHeight; canvas.getContext('2d').drawImage(video, 0, 0); stream.getVideoTracks()[0].stop(); div.remove(); return canvas.toDataURL('image/jpeg', quality); } ''') display(js) data = eval_js('takePhoto({})'.format(quality)) binary = b64decode(data.split(',')[1]) with open(filename, 'wb') as f: f.write(binary) return filename # 2. Основной цикл программы try: print("Камера запускается...") filename = take_photo() # Загружаем фото в OpenCV img = cv2.imread(filename) gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY) # Применяем порог (threshold), чтобы оставить только четкие линии (как цифры) # Это создаст эффект "матрицы" или сканера цифр thresh = cv2.adaptiveThreshold(gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 11, 2) # Ищем контуры, которые похожи на символы contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE) for cnt in contours: x, y, w, h = cv2.boundingRect(cnt) # Если объект похож по размеру на цифру, рисуем рамку if h > 10 and w > 5: cv2.rectangle(img, (x, y), (x + w, y + h), (0, 255, 0), 2) cv2.putText(img, "NUM", (x, y-5), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 1) # Сохраняем и показываем результат cv2.imwrite('result.jpg', img) from google.colab.patches import cv2_imshow print("Результат обработки:") cv2_imshow(img) except Exception as err: print(str(err))
